/**
 * Message structure mods in hard x servers conform to (WIP, many messages visible in the servers don't follow this):
 * [source] <client> Message goes here!
 * 
 * Colours:
 * - {default} -> default font, least important text
 * - {green} -> bold font
 * - {olive} -> Highlighted text, most important info of the message
 * - {blue} -> survivor client name
 * - {red} -> infected client name
 * 
 * [source] should usually refer to the action or interaction triggering the message, as that's 
 * more intutive. If the message can't be assicated with this specific of an event, then it should
 * refer to the overall system or plugin generating the message.
 * 
 * <client> is an optional field, and should exist when the message refers to a specific client.
 * 
 * Brackets containing the above should be bold. A "source" tag should be highlighted with {blue}. 
 * Hard X servers will always have a survivor alive when a message displays, so {blue} works fine.
 * Choosing this allows the prefix tag to stand out while keeping {olive} reserved for important
 * words within the actual message
 * 
 * If a 2nd client is mentioned, they should just be a part of the regular, non-bracketed text but
 * with the appropriate colour ({blue} or {red}).
 * 
 * This file is for all prefix tags from any plugin conforming to this message structure. Kept
 * all in one file as it's easy to keep future formatting changes consistent as it's impossible
 * to not miss one this way
 */
"Phrases"
{
	/* <client> tag */
	"tag_client_survivor"
	{
		/**
		 * 1: survivor client index (%N)
		 */
		"#format"	"{1:N}"
		"en"		"{green}<{blue}{1}{green}>"
	}
	"tag_client_infected"
	{
		/**
		 * 1: infected client index (%N)
		 */
		"#format"	"{1:N}"
		"en"		"{green}<{red}{1}{green}>"
	}

	/* [source] tag */
	"tag_source_hardx"
	{
		/**
		 * from hard x mod. Used for messages related to updates to the gamemode rules (like changing max specials)
		 */
		"en"	"{green}[{blue}hardx{green}]"
	}
	"tag_source_debug"
	{
		/**
		 * message from a debug command
		 */
		"en"	"{green}[{blue}debug{green}]"
	}
	"tag_source_ammo"
	{
		/**
		 * message regarding remaining ammo
		 */
		"en"	"{green}[{blue}ammo{green}]"
	}
}
